// Types
type Address = {
  village: string;
  postOffice: string;
  upazila: string;
  district: string;
};

type Parents = {
  fatherName: string;
  motherName: string;
  guardianPhone: string;
};

type Gender = "Male" | "Female" | "Others";
type Group = "Science" | "Commerce" | "Arts";
enum Role {
  Student,
  Teacher,
  Admin,
}

type Student = {
  id: number;
  name: string;
  phone: string;
  roll: number;
  class: string;
  group: Group;
  subjects: string;
  gender: Gender;
  address: Address;
  parents: Parents;
  role: Role;
  isActive: boolean;
};
